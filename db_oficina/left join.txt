SELECT V_UNIDADE.`DESC_UNIDADE`,  equipamento.`EQP_DMR`, TARGET.`EQP_SRN`  FROM v_conjunto_monitorado 
LEFT JOIN V_UNIDADE ON(v_conjunto_monitorado.`TRG_ID`= V_UNIDADE.`TRG_ID`)
LEFT JOIN TARGET ON (v_conjunto_monitorado.`TRG_ID`= target.`TRG_ID`)
LEFT JOIN equipamento ON (equipamento.`EQP_SRN`=TARGET.`EQP_SRN`)


SELECT V_UNIDADE.`DESC_UNIDADE`,

SELECT V_UNIDADE.`DESC_UNIDADE`, v_unidade.`PES_ID` AS 'Cliente', TARGET.`EQP_SRN` AS "Serial", target.`TRG_ID` , equipamento.`EQP_DMR` 'TERMINAL SATELITAL', equipamento.`EQP_TIP` 'Tipo' FROM v_conjunto_monitorado
LEFT JOIN TARGET ON (v_conjunto_monitorado.`TRG_ID`= target.`TRG_ID`)
LEFT JOIN V_UNIDADE ON(v_unidade.`TRG_ID`= target.`TRG_ID`)
LEFT JOIN equipamento ON (equipamento.`EQP_SRN`= TARGET.`EQP_SRN`)

SELECT V_UNIDADE.`DESC_UNIDADE` , CLIENTE.`CLI_COD`, TARGET.`TRG_ID`, TARGET.`TRG_DSC`, EQUIPAMENTO.`EQP_TIP` AS 'TIPO', EQUIPAMENTO.`EQP_SRN` AS 'SERIAL', EQUIPAMENTO.`EQP_DMR` 'ANTENA', EQUIPAMENTO.`PES_ID` 'CLIENTE' FROM TARGET 
LEFT JOIN EQUIPAMENTO ON (EQUIPAMENTO.`EQP_SRN` = TARGET.`EQP_SRN`)
LEFT JOIN V_UNIDADE ON (V_UNIDADE.`TRG_ID` = TARGET.`TRG_ID`)
LEFT JOIN CLIENTE ON (CLIENTE.`PES_ID` = TARGET.`PES_ID`)
-- LEFT JOIN EQUIPAMENTO ON (EQUIPAMENTO.`PES_ID` = CLIENTE.`PES_ID`)

SELECT CLIENTE.`CLI_COD`, EQUIPAMENTO.`EQP_TIP`, EQUIPAMENTO.`EQP_SRN`, EQUIPAMENTO.`EQP_DMR`, EQUIPAMENTO.`PES_ID` FROM CLIENTE
LEFT JOIN EQUIPAMENTO ON (CLIENTE.`PES_ID` = EQUIPAMENTO.`PES_ID`)
